console.log("Числа фибаначи");

function fibanachi(n) {
if (n === 0) {
return 0;

console.log("Сумма");

function sumOfNumb(numb) {
numb = Math.abs(numb);
if (numb < 10) {

return (numb % 10) + sumOfNumb(Math.floor(numb / 10));
